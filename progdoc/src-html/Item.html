002/**
003 *  An Item is an object usable in the game.
004 *  Effect may vary between each Items.
005 *
006 * @author Erwan SANGCHANMAHOLA
007 * @version 0.2
008 */
009public class Item
010{
011    
012    private int aItemW;
013    private String aItemD;
014    private String aItemN;
015    /**
016     * Create Item with name, description and weight
017     * @param pN Item's Name.
018     * @param pD Item's Description.
019     * @param pW Item's Weight.
020     */
021    
022
023    public Item(final String pN, final String pD, final int pW )
024    {
025        this.aItemW  = pW;
026        this.aItemD = pD;
027        this.aItemN = pN;
028    } 
029    /**
030     * @return The Item's weight.
031     */
032    public int getWeight()
033    {
034        return this.aItemW;
035    } 
036    /**
037     * @return The Item's Description.
038     */
039    public String getDescription()
040    {
041        return this.aItemD;
042    } 
043    /**
044     * @return The Item's Name.
045     */
046    public String getName()
047    {
048        return this.aItemN;
049    } 
050    /**
051     * @return String with the Name, Description, Weight of the Item.
052     */
053    @Override public String toString(){
054        return this.aItemN+": "+this.aItemD+" ("+this.aItemW+" g"+")";
055    }
056    
057}
